create table community (
    seq number not null,                -- 글번호

    name varchar2(40) not null,         -- 이름
    subject varchar2(255) not null,     -- 제목
    content varchar2(4000) not null,    -- 내용
    hit number default 0,               -- 조회수
    logtime date default sysdate,       -- 작성일
    primary key(seq)
);
